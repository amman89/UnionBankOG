CREATE OR REPLACE FUNCTION EXTRACT_NUMBER(IN_NUMBER VARCHAR2) RETURN VARCHAR2 IS
BEGIN
  RETURN REGEXP_REPLACE(IN_NUMBER, '[^[:digit:]]', '');
END;

CREATE OR REPLACE PUBLIC SYNONYM EXTRACT_NUMBER FOR EXTRACT_NUMBER;
GRANT EXECUTE ON EXTRACT_NUMBER TO APP_ADMIN_ROLE;
GRANT EXECUTE ON EXTRACT_NUMBER TO APP_USR_ROLE;
